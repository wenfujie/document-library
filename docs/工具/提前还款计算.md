# 提前还款计算器

在下方输入你的贷款数据，实时查看提前还款带来的期数与利息变化。

<script setup>
import { ref, computed } from 'vue'

// 响应式输入（等价于 data 中的 4 个属性）
const monthPay = ref(1900)
const debt = ref(400000)
const repayment = ref(100000)
const yearRate = ref(0.03)

// 计算属性：computedGold
const computedGold = computed(() => {
  const M = Number(monthPay.value) || 0
  const P = Math.max(0, Number(debt.value) || 0)
  const R = Math.max(0, Number(repayment.value) || 0)
  const Y = Number(yearRate.value) || 0

  const r = Y / 12
  const P2 = Math.max(0, P - R)

  if (M <= 0) return { error: '请输入大于 0 的月供金额' }
  if (P <= 0) return { error: '请输入大于 0 的欠款本金' }
  if (Y < 0) return { error: '年利率不能为负数' }
  if (r > 0 && M <= P2 * r) return { error: '月供不足以覆盖当期利息，请提高月供或降低本金' }
  if (r > 0 && M <= P * r) return { error: '月供不足以覆盖当期利息，请提高月供或降低本金' }

  const periods = (principal) => {
    if (r === 0) return Math.ceil(principal / M)
    const denom = M - principal * r
    if (denom <= 0) return Infinity
    const n = Math.log(M / denom) / Math.log(1 + r)
    return Math.max(0, Math.ceil(n))
  }

  const n1 = periods(P)
  const n2 = periods(P2)
  if (!isFinite(n1) || !isFinite(n2)) return { error: '当前参数无法计算出合理期数，请调整后重试' }

  const acc1 = M * n1 - P
  const acc2 = M * n2 - P2

  return {
    principalBefore: P,
    principalAfter: P2,
    accrualsBefore: Math.max(0, acc1),
    accrualsAfter: Math.max(0, acc2),
    accrualsSaved: Math.max(0, acc1 - acc2),
    periodBefore: n1,
    periodAfter: n2,
    periodsSaved: Math.max(0, n1 - n2)
  }
})

// 工具函数
const money = (n) => {
  const v = Number(n)
  if (!isFinite(v)) return '-'
  return v.toLocaleString('zh-CN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })
}
</script>
